void get_dht(Dht11 &s) {
  s.h = dht.readHumidity();
  s.t = dht.readTemperature();

  if (isnan(s.h) || isnan(s.t)) {
    s.h = -1;
    s.t = -1;
    s.hic = -1;
    Serial.println("Lecture DHT invalide");
    return;
  }

  s.hic = dht.computeHeatIndex(s.t, s.h, false);
  Serial.print("T=");
  Serial.print(s.t);
  Serial.print(" H=");
  Serial.print(s.h);
  Serial.print(" HIC=");
  Serial.println(s.hic);
}

void config_buzzer(Buzzer &b) {
  analogWriteResolution(8);
  pinMode(b.pin, OUTPUT);
  analogWrite(b.pin, 0);
  Serial.print("Buzzer init pin=");
  Serial.println(b.pin);
}

void buzzer_start(Buzzer &b, uint32_t duration) {
  b.start = millis();
  b.duration = duration;
  b.active = true;
  analogWrite(b.pin, b.level);
  Serial.print("Start freq=");
  Serial.print(b.freq);
  Serial.print(" level=");
  Serial.print(b.level);
  Serial.print(" duration=");
  Serial.println(b.duration);
}

void buzzer_update(Buzzer &b) {
  if (!b.active) return;

  if (millis() - b.start >= b.duration) {
    analogWrite(b.pin, 0);
    b.active = false;
    b.start = 0;
    Serial.println("Stop buzzer");
  }
}

    Serial.println("=== TEST HMI ===");
    if (ok) Serial.println("IHM OK");
    else {
        Serial.println("Erreur init OLED");

void loop() {
    uint32_t now = millis();

    // Mise à jour périodique de l'IHM
    ihm_update(now);

    // Fausse évolution des mesures toutes les 1 s
    if (now - lastDataUpdate >= 1000) {
        lastDataUpdate = now;

        data.temperature += 0.2f;
        if (data.temperature > 36.0f) data.temperature = 24.0f;

        data.humidity += 0.5f;
        if (data.humidity > 80.0f) data.humidity = 50.0f;

        data.rainValue += 50;
        if (data.rainValue > 1500) data.rainValue = 0;

        data.windCount += 1;
        if (data.windCount > 20) data.windCount = 0;

        ihm_set_data(data);

        Serial.print("Temp=");
        Serial.print(data.temperature);
        Serial.print(" Hum=");
        Serial.print(data.humidity);
        Serial.print(" Rain=");
        Serial.print(data.rainValue);
        Serial.print(" Wind=");
        Serial.println(data.windCount);
    }

    // Scénario d'états toutes les 8 s
    if (now - lastScenarioStep >= 8000) {
        lastScenarioStep = now;
        scenario++;

        switch (scenario % 6) {
            case 0:
                data.wifiConnected = false;
                data.wifiRssi = -90;
                ihm_set_data(data);
                ihm_set_state(IHM_NORMAL);
                Serial.println("[SCENARIO] NORMAL / WiFi OFF");
                break;

            case 1:
                data.wifiConnected = true;
                data.wifiRssi = -58;
                ihm_set_data(data);
                ihm_set_state(IHM_NORMAL);
                Serial.println("[SCENARIO] NORMAL / WiFi ON");
                break;

            case 2:
                ihm_set_state(IHM_RAIN);
                Serial.println("[SCENARIO] ALERTE PLUIE");
                break;

            case 3:
                ihm_set_state(IHM_WIND);
                Serial.println("[SCENARIO] ALERTE VENT");
                break;

            case 4:
                ihm_set_state(IHM_RAIN_WIND);
                Serial.println("[SCENARIO] ALERTE PLUIE + VENT");
                break;

            case 5:
                ihm_set_state(IHM_OVERHEAT);
                Serial.println("[SCENARIO] ALERTE SURCHAUFFE");
                break;
        }
    }
}
1. Mode “Ventilo extraction” (refroidissement global)
Condition d’activation : temp >= SEUIL_TEMP_SURCHAUFFE → état “SURCHAUFFE”.

Action FSM dans cet état :

ventilo = ON pour extraire l’air chaud du boîtier (flux global).

Le volet reste piloté par la logique météo (moteur), le ventilo tourne même si volet fermé.

Retour à l’état normal : temp < SEUIL_TEMP_SURCHAUFFE - hystérésis → ventilo = OFF.

2. Mode “Ventilo ciblé composants” (refroidissement local)
Condition d’activation : même état “SURCHAUFFE” (même seuil).

Action FSM dans cet état :

ventilo = ON pour souffler directement sur les composants critiques (zone chaude).

Toujours indépendant : même si l’état météo impose volet fermé, le ventilo reste actif sur l’électronique.

Retour : pareil, dès que la température redescend sous le seuil bas, ventilo = OFF.

void loop() {
    // Exemple simple : balayage 0 -> 90 -> 180 -> 0
    static uint8_t step = 0;
    static uint32_t lastChange = 0;
    uint32_t now = millis();

    if (now - lastChange >= 2000) {
        lastChange = now;
        switch (step) {
            case 0: servo_set_angle(volet, 0);   Serial.println("Angle 0");   break;
            case 1: servo_set_angle(volet, 90);  Serial.println("Angle 90");  break;
            case 2: servo_set_angle(volet, 180); Serial.println("Angle 180"); break;
        }
        step = (step + 1) % 3;
    }

    servo_update(volet, now);
